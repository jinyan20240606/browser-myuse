# 2. 查看当前已有的远程仓库（通常会有 origin，指向你的仓库）
git remote -v

# 3. 添加上游仓库（替换为原开源项目的 Git 地址）
git remote add upstream https://github.com/browser-use/browser-use.git

# 4. 验证是否添加成功（会看到 upstream 对应的地址）
git remote -v

# 1. 拉取上游开源仓库的所有分支更新（不会直接合并到你的代码）
git fetch upstream

# 2. 切换到你的个人本地主开发分支
git checkout source-main

# 3. 合并上游的主分支更新到你的分支（解决冲突）
git merge upstream/main
